## Install

```bash
yarn add degen
```

## Wrap root app and import stylesheet

```tsx
import { ThemeProvider } from 'degen'
import 'degen/dist/style.css'

const App = () => {
  return <ThemeProvider>{children}</ThemeProvider>
}
```

## Import and use components

```tsx
import { Button, Heading, IconLockClosed, Stack } from 'degen'
```

```tsx live=true expand=true
<Stack
```
