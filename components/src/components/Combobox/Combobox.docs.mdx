---
title: Combobox
description: Input for filtering and selecting an option
---

```tsx
import { Combobox } from 'degen'
```

```tsx live=true expand=true
<Combobox
  filter={(options, inputValue) =>
    matchSorter(options, inputValue, { keys: ['label'] })
  }
  label="Nouns"
  options={options}
  placeholder="Search Nouns…"
/>
```

  filter={(options, inputValue) =>
    matchSorter(options, inputValue, { keys: ['label'] })
  }
  label="Nouns"
