---
title: Combobox
description: Input for filtering and selecting an option
---

```tsx
import { Combobox } from 'degen'
```

```tsx live=true expand=true
<Combobox
  filter={(options, inputValue) =>
    matchSorter(options, inputValue, { keys: ['label'] })
  }
  label="Nouns"
  options={options}
  placeholder="Search Nouns…"
  value={getState('createValue')}
  onChange={(value) => setState('createValue', value)}
/>
```

## Create

```tsx live=true
<Combobox
  createText={(value) => `Create ${value} Noun`}
  filter={(options, inputValue) =>
    matchSorter(options, inputValue, { keys: ['label'] })
  }
  label="Nouns"
  loading
  options={options}
  placeholder="Search Nouns…"
  value={getState('createValue')}
  onChange={(value) => setState('createValue', value)}
  onCreate={(value) => alert(`Create ${value} Noun`)}
/>
```
